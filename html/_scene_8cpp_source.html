139 void Scene::loadMatricesToShader(const char* _shaderName, ngl::Mat4 _M)
140 {

334 void Scene::createFramebufferObject()
335 {
336  // make two fbos: fbo[0] and fbo[1], the history ping pong buffers
337  // fbo[0] has attachments fboTexId[0] and fboDepthId[0], and similarly for fbo[1]...
338  for(int i=0;i<2;i++)
339  {
340  // create framebuffer objects, these are deleted in the dtor
341  glGenFramebuffers(1, &m_fboId[i]);
342  glBindFramebuffer(GL_FRAMEBUFFER, m_fboId[i]);
343 
344  // attatch the textures we created earlier to the FBO
345  glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, m_fboTexId[i], 0);
346 
347  // now got back to the default render context
348  glBindFramebuffer(GL_FRAMEBUFFER, 0);
349 
350  // check FBO attachment success
351  if(glCheckFramebufferStatus(GL_FRAMEBUFFER) ==
352  GL_FRAMEBUFFER_COMPLETE || !m_fboTexId[i])
353  {
354  printf("Framebuffer %d OK\n", i);
355  }
356  }
357 
358  // do the same for our current frame fbos
359  glGenFramebuffers( 1, &m_fboId[2] );
360  glBindFramebuffer( GL_FRAMEBUFFER, m_fboId[2] );
361  glFramebufferTexture2D( GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, m_fboTexId[2], 0 );
362  glFramebufferTexture2D( GL_FRAMEBUFFER, GL_DEPTH_ATTACHMENT, GL_TEXTURE_2D, m_fboDepthId, 0 );
363  glBindFramebuffer(GL_FRAMEBUFFER, 0);
364  // check FBO attachment success
365  if(glCheckFramebufferStatus(GL_FRAMEBUFFER) ==
366  GL_FRAMEBUFFER_COMPLETE || !m_fboTexId[2] || !m_fboDepthId )
367  {
368  printf("Framebuffer 2 OK\n");

381 void Scene::paintGL()
382 {
383  // grab an instance of ngl shader manager and ngl primitives
384  ngl::ShaderLib *shader=ngl::ShaderLib::instance();
385  ngl::VAOPrimitives* prim = ngl::VAOPrimitives::instance();
386  // shader program id
387  GLuint pid;
388 
389  // to help distinguish the ping pong fbos and their texture locations
390  int pong = 1-m_ping;
391  // the locations of the ping pong textures starts at location 4
392  int historyLocation = 4;
393 
394  //------------------CurrentFrame FBO-------------------
395 
396  // set the rendering target to the "current frame" fbo
397  glBindFramebuffer(GL_DRAW_FRAMEBUFFER, m_fboId[2]); // always fbo[2]
398 
399  // clear the current rendering target
400  glClearColor(0.2f,0.5f,0.8f,1.0f);
401  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
402  glViewport(0, 0, TEXTURE_WIDTH, TEXTURE_HEIGHT);
403 
404  // bind the current fbo's color attachment to a texture slot in memory
405  // always TEXTURE2
406  glActiveTexture(GL_TEXTURE2);
407  glBindTexture(GL_TEXTURE_2D, m_fboTexId[2]);
408 
409  // bind the current fbo's depth attachment to a texture slot in memory
410  glActiveTexture(GL_TEXTURE3);
411  glBindTexture(GL_TEXTURE_2D, m_fboDepthId);
412 
413  // draw the scene to the current rendering target
414  drawSceneGeometry(prim);
415 
416  if(m_TAA)
417  {
418  //-------------------History FBO--------------------
419 
420  // bind to one of the history ping-pong buffers (either fbo[1] or fbo[0])
421  glBindFramebuffer(GL_FRAMEBUFFER, m_fboId[m_ping]);
422  // clear
423  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
424 
425  // bind the blend fbo's color attachment to a texture slot in memory
426  // (if it's fbo[0] to TEXTURE4, if it's fbo[1] to TEXTURE5)
427  // because of the texture's location, it will become previous history in next frame
428  glActiveTexture(GL_TEXTURE0 + historyLocation + m_ping);
429  glBindTexture(GL_TEXTURE_2D, m_fboTexId[m_ping]);
430 
431  // use the TAA shader to draw our image onto the screen-oriented plane
432  (*shader)["TAA"]->use();
433  pid = shader->getProgramID("TAA");
434 
435  // send subroutine uniforms to the shader
436  //GLuint *indices;
437  GLuint nbrClamp_subroutine_index;
438  if(m_subNbrClamp)
439  {
440  nbrClamp_subroutine_index = glGetSubroutineIndex(pid, GL_FRAGMENT_SHADER, "nbrClampOn" );
441  }
442  else
443  {
444  nbrClamp_subroutine_index = glGetSubroutineIndex(pid, GL_FRAGMENT_SHADER, "nbrClampOff" );
445  }
446  glUniformSubroutinesuiv(GL_FRAGMENT_SHADER, 1, &nbrClamp_subroutine_index);
447 
448  // send textures to the shader
449  glUniform1i(glGetUniformLocation(pid, "_currentFrameTex"), 2);
450  glUniform1i(glGetUniformLocation(pid, "_currentDepthTex"), 3);
451  glUniform1i(glGetUniformLocation(pid, "_previousFrameTex"), historyLocation + pong); //either 5 or 4
452 
453  ngl::Vec3 currPos(m_view[12],m_view[13],m_view[14]);
454  ngl::Vec3 prevPos(m_viewPrev[12],m_viewPrev[13],m_viewPrev[14]);
455 
456  // send other parameter uniforms
457  shader->setUniform("_windowSize", ngl::Vec2(TEXTURE_WIDTH,TEXTURE_HEIGHT) );
458  shader->setUniform("_start", m_start );
459  shader->setUniform("_viewInverse", m_view.inverse() );
460  shader->setUniform("_viewPrev", m_viewPrev );
461  shader->setUniform("_deltaDist", currPos-prevPos );
462  shader->setUniform("_projection", m_projection );
463  shader->setUniform("_projectionInverse", m_projection.inverse() );
464  // multiply the jitter offset by 0.5 to fit the canonical view volume
465  shader->setUniform("_jitter", m_jitter.getOffset()*ngl::Vec2(0.5,0.5) );
466 
467  // draw the plane
468  drawScreenOrientedPlane(pid,prim);
469  }
470 
471  //--------------------Default FBO----------------------
472 
473  // bind to the default framebuffer
474  glBindFramebuffer(GL_FRAMEBUFFER, 0);
475  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
476  glViewport(0, 0, TEXTURE_WIDTH, TEXTURE_HEIGHT);
477 
478  // use the Texture shader to draw our image onto the screen-oriented plane
479  (*shader)["Zoom"]->use();
480  pid = shader->getProgramID("Zoom");
481 
482  // send zoom subroutine to the shader
483  GLuint zoom_subroutine_index;
484  if(m_cam.zoom)
485  {
486  zoom_subroutine_index = glGetSubroutineIndex(pid, GL_FRAGMENT_SHADER, "on" );
487  }
488  else
489  {
490  zoom_subroutine_index = glGetSubroutineIndex(pid, GL_FRAGMENT_SHADER, "off" );
491  }
492  glUniformSubroutinesuiv(GL_FRAGMENT_SHADER, 1, &zoom_subroutine_index);
493 
494  // send the history output texture to the GPU
495  glUniform1i(glGetUniformLocation(pid, "_frameTex"), (historyLocation+m_ping)*m_TAA + !m_TAA*2 ); // either 3 or 4, or 1 (current frame, no TAA)
496 
497  // send parameters for zoom
498  shader->setUniform("_mousePos", ngl::Vec2(m_cam.cursorX, -m_cam.cursorY) );
499  shader->setUniform("_windowSize", ngl::Vec2(TEXTURE_WIDTH,TEXTURE_HEIGHT) );
500  // the dimensions of the texture when zoomed-in
501  shader->setUniform("_zoomSize", ngl::Vec2(TEXTURE_WIDTH*m_cam.zoomScale,
502  TEXTURE_HEIGHT*m_cam.zoomScale) );
503  shader->setUniform("_zoomSquare", m_cam.zoomSquare );
504 
505  // draw the plane
506  drawScreenOrientedPlane(pid, prim);
507 
508  //-----------------------------------------------------
509 
510  // switch the rendering target index (from 0 -> 1, or from 1 -> 0)
511  m_ping = pong;
512 
513  // store the current view matrix for next frame's "previous" view matrix
514  m_viewPrev = m_view;
515 
516  // no longer the first frame
517  m_start = false;
518 
519  // render the text that displays the current TAA settings
520  //renderText();
521 
522  // update the image
523  update();
